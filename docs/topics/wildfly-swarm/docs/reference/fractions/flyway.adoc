= Flyway


.Maven Coordinates
[source,xml]
----
<dependency>
  <groupId>org.wildfly.swarm</groupId>
  <artifactId>flyway</artifactId>
</dependency>
----

.Configuration

swarm.flyway.jdbc-password:: 
JDBC connection password

swarm.flyway.jdbc-url:: 
JDBC connection URL

swarm.flyway.jdbc-user:: 
JDBC connection user name
